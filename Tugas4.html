import numpy as np
from scipy.stats import norm

# Data training
train_data = np.array([
    [5.1, 3.5, 'Iris-setosa'],
    [6.4, 2.9, 'Iris-versicolor'],
    [4.9, 3.0, 'Iris-setosa'],
    [6.9, 3.1, 'Iris-versicolor'],
    [5.0, 3.6, 'Iris-setosa'],
    [5.5, 2.3, 'Iris-versicolor'],
    [5.4, 3.9, 'Iris-setosa'],
    [6.5, 2.8, 'Iris-versicolor'],
    [5.7, 2.8, 'Iris-versicolor'],
    [6.3, 2.7, 'Iris-versicolor']
], dtype=object)

# Data uji
test_data = np.array([5.5, 3.0])

# Hitung prior
classes = np.unique(train_data[:, 2])
priors = {cls: np.sum(train_data[:, 2] == cls) / len(train_data) for cls in classes}

# Hitung mean dan varians
stats = {}
for cls in classes:
    subset = train_data[train_data[:, 2] == cls][:, :2].astype(float)
    stats[cls] = {'mean': np.mean(subset, axis=0), 'var': np.var(subset, axis=0, ddof=0)}

# Hitung likelihood dan posterior probability
posteriors = {}
for cls in classes:
    likelihood = np.prod(norm.pdf(test_data, loc=stats[cls]['mean'], scale=np.sqrt(stats[cls]['var'])))
    posteriors[cls] = likelihood * priors[cls]

# Prediksi kelas
predicted_class = max(posteriors, key=posteriors.get)

# Tampilkan hasil perhitungan
def print_results():
    print("Prior Probabilities:")
    for cls in classes:
        print(f"P({cls}) = {priors[cls]:.4f}")

    print("\nMean and Variance:")
    for cls in classes:
        print(f"{cls}: Mean = {stats[cls]['mean']}, Variance = {stats[cls]['var']}")

    print("\nLikelihood Calculations:")
    for cls in classes:
        likelihood = np.prod(norm.pdf(test_data, loc=stats[cls]['mean'], scale=np.sqrt(stats[cls]['var'])))
        print(f"P(X | {cls}) = {likelihood:.6f}")

    print("\nPosterior Probabilities:")
    for cls in classes:
        print(f"P({cls} | X) = {posteriors[cls]:.6f}")

    print(f"\nPredicted Class: {predicted_class}")

print_results()
