Local Outlier Factor (LOF)#

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.neighbors import LocalOutlierFactor

# Data contoh (10 data dengan 2 fitur)
data = np.array([
    [5.1, 3.5], [4.9, 3.0], [4.7, 3.2], [4.6, 3.1], [5.0, 3.6],
    [5.4, 3.9], [4.6, 3.4], [5.0, 3.4], [4.4, 2.9], [4.9, 3.1]
])

# Inisialisasi model LOF dengan k=3
lof = LocalOutlierFactor(n_neighbors=3, metric="euclidean")

# Fit model dan dapatkan skor LOF (semakin negatif, semakin outlier)
lof_scores = lof.fit_predict(data)
lof_values = -lof.negative_outlier_factor_

# Buat DataFrame untuk melihat hasilnya
df = pd.DataFrame(data, columns=["SL", "SW"])
df.insert(0, "ID", range(1, 11))  # Menambahkan kolom ID dari 1-10
df["LOF"] = lof_values
df["Outlier"] = ["Ya" if x > 1.2 else "Tidak" for x in lof_values]

# Tampilkan hasil
print(df)

# Visualisasi data
plt.figure(figsize=(8, 6))
sns.scatterplot(x=df["SL"], y=df["SW"], hue=df["Outlier"], palette={"Ya": "red", "Tidak": "blue"}, s=100)
plt.title("Deteksi Outlier dengan LOF")
plt.xlabel("SL")
plt.ylabel("SW")
plt.show()
