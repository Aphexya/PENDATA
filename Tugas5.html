import pandas as pd
import numpy as np
from sklearn.datasets import load_iris
from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_score
from scipy.stats import mode
import matplotlib.pyplot as plt

# Load dataset iris
iris = load_iris()
X = pd.DataFrame(iris.data, columns=iris.feature_names)
y = iris.target  # label asli

# Ambil centroid dari data ke-1 dan 110 (indeks Python: 0 dan 109)
init_indices = [0, 109]
manual_centroids = X.iloc[init_indices].values

# KMeans dengan 2 cluster dan centroid manual
kmeans = KMeans(n_clusters=2, init=manual_centroids, n_init=1, random_state=42)
labels = kmeans.fit_predict(X)

# Hitung SSE dan Silhouette
sse = kmeans.inertia_
sil_score = silhouette_score(X, labels)

# Akurasi Clustering
def calculate_accuracy(true_labels, cluster_labels):
    labels_map = {}
    for cluster in np.unique(cluster_labels):
        mask = cluster_labels == cluster
        mapped_label = mode(true_labels[mask], keepdims=True).mode[0]
        labels_map[cluster] = mapped_label
    mapped_predictions = np.array([labels_map[cl] for cl in cluster_labels])
    accuracy = np.mean(mapped_predictions == true_labels)
    return accuracy

akurasi = calculate_accuracy(y, labels)

# Cetak hasil
print("=== Hasil Clustering KMeans (k=2) ===")
print(f"SSE (Inertia): {sse:.2f}")
print(f"Silhouette Score: {sil_score:.4f}")
print(f"Akurasi Clustering: {akurasi:.4f}")
print("Jumlah iterasi hingga konvergen:", kmeans.n_iter_)

# Tampilkan centroid dan jumlah anggota
X_temp = X.copy()
X_temp['Cluster'] = labels
centroids = X_temp.groupby('Cluster').mean()
counts = X_temp['Cluster'].value_counts().sort_index()
centroids['Anggota'] = counts.values
print("\nCentroid dan Jumlah Anggota:")
print(centroids)

# Visualisasi hasil clustering
plt.figure(figsize=(6, 4))
plt.scatter(X.iloc[:, 0], X.iloc[:, 1], c=labels, cmap='viridis', label='Data')
plt.scatter(manual_centroids[:, 0], manual_centroids[:, 1], color='red', marker='X', s=200, label='Initial Centroids')
plt.title("Hasil Clustering KMeans (k=2)")
plt.xlabel(iris.feature_names[0])
plt.ylabel(iris.feature_names[1])
plt.legend()
plt.tight_layout()
plt.show()
