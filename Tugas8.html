from sklearn.metrics import silhouette_score

# Load dataset Iris dari sklearn
iris = load_iris()

feature_names = iris.feature_names
target = iris.target
target_names = iris.target_names

# Buat DataFrame
df = pd.DataFrame(X_data, columns=[f.replace(' (cm)', '').replace(' ', '_') for f in feature_names])
df['Class'] = [target_names[i] for i in target]

# Fungsi clustering dan kategori tanpa menyimpan kolom cluster
def cluster_dan_kategori(df, kolom, k, abjad_kategori):
    X = df[[kolom]].values
    kmeans = KMeans(n_clusters=k, n_init=100, random_state=42)
    kmeans.fit(X)
    centroids = kmeans.cluster_centers_

    # Evaluasi
    sil_score = silhouette_score(X, kmeans.labels_)

    # Kategori berdasarkan centroid terurut
    sorted_centroids = np.sort(centroids.flatten())
    def tentukan_kategori(nilai):
        jarak = [abs(nilai - c) for c in sorted_centroids]
        idx_terdekat = np.argmin(jarak)
        return abjad_kategori[idx_terdekat]

    kategori_col = f'kategori_{kolom}'
    df[kategori_col] = df[kolom].apply(tentukan_kategori)

    # Tampilkan info
    print(f"\n=== Kategori untuk '{kolom}' (K={k}) ===")
    for i, c in enumerate(centroids):
        print(f"Centroid {i}: {c[0]:.4f}")
    print(f"Silhouette Score : {sil_score:.4f}")
    print("Contoh hasil kategori:")
    print(df[[kolom, kategori_col]].head())
    return df

#✨ Langkah-langkah Diskritisasi dengan K-Means:

  1. Ambil 1 kolom numerik, misalnya sepal_length.

  2. Jalankan K-Means dengan jumlah klaster k tertentu.

  3. K-Means akan mencari k buah centroid dan memberi label 0,1,2,… pada setiap data.

  4. Kita urutkan centroid, dan beri nama kategori berdasarkan urutan (misalnya A, B, C, …).

  5. Ulangi proses ini untuk setiap fitur numerik.

# PROSES DISKRITISASI DENGAN K-MEANS

def cluster_dan_kategori(df, kolom, k, abjad_kategori):
    """
    Fungsi untuk melakukan diskritisasi menggunakan K-Means clustering

    Parameters:
    - df: DataFrame
    - kolom: nama kolom yang akan didiskritisasi
    - k: jumlah cluster (kategori)
    - abjad_kategori: list kategori (A, B, C, D)

    Returns:
    - df: DataFrame dengan kolom kategori baru
    """
    X = df[[kolom]].values

    # Lakukan K-Means clustering
    kmeans = KMeans(n_clusters=k, n_init=100, random_state=42)
    kmeans.fit(X)
    centroids = kmeans.cluster_centers_

    # Evaluasi clustering dengan silhouette score
    sil_score = silhouette_score(X, kmeans.labels_)

    # Urutkan centroids untuk konsistensi kategori
    sorted_centroids = np.sort(centroids.flatten())

    def tentukan_kategori(nilai):
        """Tentukan kategori berdasarkan centroid terdekat"""
        jarak = [abs(nilai - c) for c in sorted_centroids]
        idx_terdekat = np.argmin(jarak)
        return abjad_kategori[idx_terdekat]

    # Buat kolom kategori baru
    kategori_col = f'kategori_{kolom}'
    df[kategori_col] = df[kolom].apply(tentukan_kategori)

    # Tampilkan informasi clustering
    print(f"\n=== DISKRITISASI '{kolom}' (K={k}) ===")
    print("Centroids yang ditemukan:")
    for i, c in enumerate(sorted_centroids):
        print(f"  Kategori {abjad_kategori[i]}: {c:.4f}")
    print(f"Silhouette Score: {sil_score:.4f}")

    # Tampilkan distribusi kategori
    distribusi = df[kategori_col].value_counts().sort_index()
    print("Distribusi kategori:")
    for kategori, count in distribusi.items():
        print(f"  {kategori}: {count} sampel")

    return df

def evaluate_classification(X, y, data_type, test_size=0.3, random_state=42):
    """
    Fungsi untuk melakukan klasifikasi dan evaluasi
    """
    print(f"\n{'='*50}")
    print(f"KLASIFIKASI PADA DATA {data_type.upper()}")
    print('='*50)

    # Split data
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=test_size, random_state=random_state, stratify=y
    )

    print(f"Training set: {X_train.shape[0]} sampel")
    print(f"Testing set: {X_test.shape[0]} sampel")

    # Initialize classifiers
    nb_classifier = GaussianNB()
    dt_classifier = DecisionTreeClassifier(random_state=42)

    classifiers = {
        'Naive Bayes': nb_classifier,
        'Decision Tree': dt_classifier
    }

    results = {}

    for name, clf in classifiers.items():
        print(f"\n--- {name} ---")

        # Training
        clf.fit(X_train, y_train)

        # Prediction
        y_pred = clf.predict(X_test)

        # Evaluation
        accuracy = accuracy_score(y_test, y_pred)
        results[name] = accuracy

        print(f"Accuracy: {accuracy:.4f}")
        print("\nClassification Report:")
        print(classification_report(y_test, y_pred))

        # Confusion Matrix
        cm = confusion_matrix(y_test, y_pred)
        print("Confusion Matrix:")
        print(cm)

    return results
