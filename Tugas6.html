#Pengertian Fuzzy C Means Fuzzy C-Means (FCM) adalah sebuah algoritma pengelompokan (clustering) yang memungkinkan setiap titik data untuk menjadi bagian dari beberapa klaster dengan derajat keanggotaan tertentu.Sebuah algoritma yg sama2 dikelompok sama seperti Kmeans

Perbedaan fuzzy Cmeans dan Kmeans :

  • K-Means: setiap data hanya boleh menjadi anggota satu cluster.

  • Fuzzy C-Means: setiap data bisa menjadi anggota dari beberapa cluster sekaligus, dengan derajat keanggotaan (nilai antara 0 dan 1).

import numpy as np

# Data xi (x1, x2)
data = np.array([
    [1, 2],
    [2, 3],
    [3, 4],
    [6, 7],
    [7, 8]
])

# Parameter
c = 2                     # Jumlah kluster
m = 2.0                   # Fuzzifier
epsilon = 0.001           # Ambang batas konvergensi
max_iter = 100            # Maksimum iterasi
n = data.shape[0]         # Jumlah data
d = data.shape[1]         # Dimensi fitur

# Matriks keanggotaan awal U (manual) – jumlah tiap kolom harus = 1
U = np.array([
    [0.8, 0.6, 0.5, 0.2, 0.1],
    [0.2, 0.4, 0.5, 0.8, 0.9]
])

def calculate_cluster_centers(U, data, m):
    """Hitung pusat cluster v"""
    um = U ** m
    v = (um @ data) / np.sum(um, axis=1, keepdims=True)
    return v

def update_membership_matrix(U, v, data, m):
    """Update matriks keanggotaan U"""
    dist = np.zeros((c, n))
    for i in range(c):
        for j in range(n):
            dist[i][j] = np.linalg.norm(data[j] - v[i]) + 1e-10  # hindari pembagian 0

    for i in range(c):
        for j in range(n):
            denom = sum((dist[i][j] / dist[k][j]) ** (2 / (m - 1)) for k in range(c))
            U[i][j] = 1 / denom
    return U

def calculate_objective_function(U, v, data, m):
    """Hitung nilai fungsi objektif J"""
    J = 0
    for i in range(c):
        for j in range(n):
            dist_ij = np.linalg.norm(data[j] - v[i])
            J += (U[i][j] ** m) * (dist_ij ** 2)
    return J

# Iterasi
for iteration in range(max_iter):
    print(f"\n=== Iterasi {iteration + 1} ===")

    U_old = U.copy()

    v = calculate_cluster_centers(U, data, m)
    print("Pusat cluster v:")
    print(v)

    U = update_membership_matrix(U, v, data, m)
    print("Matriks keanggotaan U (derajat keanggotaan tiap data ke tiap cluster):")
    for j in range(n):
        print(f"Data {j+1} {data[j]}: ", end='')
        for i in range(c):
            print(f"Cluster {i+1}: {U[i][j]:.4f} ", end='')
        print()

    J = calculate_objective_function(U, v, data, m)
    print(f"Nilai fungsi objektif J: {J:.6f}")

    if np.linalg.norm(U - U_old) < epsilon:
        print("\nKonvergen tercapai.")
        break

print("\n=== Hasil Akhir Clustering ===")
labels = np.argmax(U, axis=0)
for i, label in enumerate(labels):
    print(f"Data {i+1} {data[i]} termasuk cluster {label+1} dengan derajat keanggotaan {U[:, i]}")

#Perhitungan Fuzzy C-Means (FCM) pada dataset Iris

import numpy as np
from sklearn.datasets import load_iris

# Muat data Iris
iris = load_iris()
data = iris.data
n = data.shape[0]
d = data.shape[1]

# Parameter FCM
c = 3
m = 2.0
epsilon = 1e-4
max_iter = 100

# Inisialisasi matriks keanggotaan U (c x n)
U = np.random.rand(c, n)
U = U / np.sum(U, axis=0, keepdims=True)

def calculate_cluster_centers(U, data, m):
    um = U ** m
    v = (um @ data) / np.sum(um, axis=1, keepdims=True)
    return v

def update_membership_matrix(U, v, data, m):
    dist = np.zeros((c, n))
    for i in range(c):
        for j in range(n):
            dist[i, j] = np.linalg.norm(data[j] - v[i]) + 1e-10  # Hindari pembagian nol

    for i in range(c):
        for j in range(n):
            denom = sum((dist[i, j] / dist[k, j]) ** (2 / (m - 1)) for k in range(c))
            U[i, j] = 1 / denom
    return U

def calculate_objective_function(U, v, data, m):
    obj = 0.0
    for i in range(c):
        for j in range(n):
            dist_sq = np.linalg.norm(data[j] - v[i]) ** 2
            obj += (U[i, j] ** m) * dist_sq
    return obj

# Iterasi FCM
iteration = 0
while iteration < max_iter:
    print(f"\n=== Iterasi {iteration + 1} ===")

    U_old = U.copy()

    v = calculate_cluster_centers(U, data, m)
    print("Pusat klaster:")
    print(v)

    U = update_membership_matrix(U, v, data, m)

    diff = np.linalg.norm(U - U_old)
    print("Perubahan U:", diff)

    # Hitung dan tampilkan fungsi objektif
    obj = calculate_objective_function(U, v, data, m)
    print("Nilai fungsi objektif:", obj)

    if diff < epsilon:
        print("\nKonvergensi tercapai.")
        break

    iteration += 1

# Tentukan label klaster akhir
labels = np.argmax(U, axis=0)

# Tampilkan hasil
print("\n=== Hasil Klastering ===")
for i in range(n):
    print(f"Data ke-{i+1} termasuk klaster {labels[i]}")

# Tampilkan derajat keanggotaan
print("\n=== Derajat Keanggotaan Tiap Data (U) ===")
for j in range(n):
    print(f"Data ke-{j+1}: ", end="")
    for i in range(c):
        print(f"u{i+1}={U[i, j]:.4f} ", end="")
    print()
